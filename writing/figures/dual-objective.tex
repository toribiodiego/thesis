% Cuboid without right side face -- avoids stub-arrow artifact
% when the cuboid is followed by an arrow on small elements.
% Args: x, width, height, depth, front-color, top-color
\newcommand{\cuboidns}[6]{%
    \fill[#5, draw=black!35, line join=round]
        (#1, {-#3/2}, 0) -- ({#1+#2}, {-#3/2}, 0) --
        ({#1+#2}, {#3/2}, 0) -- (#1, {#3/2}, 0) -- cycle;
    \fill[#6, draw=black!35, line join=round]
        (#1, {#3/2}, 0) -- ({#1+#2}, {#3/2}, 0) --
        ({#1+#2}, {#3/2}, #4) -- (#1, {#3/2}, #4) -- cycle;
}

\begin{tikzpicture}[
    x={(1cm,0cm)}, y={(0cm,1cm)}, z={(0.35cm,0.25cm)},

    lbl/.style={font=\scriptsize, text=black!45},
]
