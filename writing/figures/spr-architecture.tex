\begin{figure}[htbp]
\centering
% Encoder colors (reuse from Nature CNN)
\definecolor{encfade}{HTML}{B8D2E6}
% SPR-specific component color
\definecolor{sprcomp}{HTML}{C2D4A8}

\begin{tikzpicture}[
    arr/.style={->, thick, >=stealth, black!55},
    lbl/.style={font=\scriptsize, text=black!55},
]

% s_t
\node[font=\small, text=black!90] at (-0.5, 0) {$s_t$};
\draw[arr] (-0.1, 0) -- (0.6, 0);

% Encoder (trapezoid: wide left, narrow right = compression)
\fill[encfade, draw=black!70, thick, line join=round]
    (0.8, -0.8) -- (0.8, 0.8) -- (2.1, 0.45) -- (2.1, -0.45) -- cycle;
\node[font=\small, text=black!85] at (1.35, 0) {$f_\theta$};

% z_t as plain text node -- arrows point to and from it
\node[font=\small, text=black!90, inner sep=2pt] (zt) at (3.0, 0) {$z_t$};
\draw[arr] (2.1, 0) -- (zt.west);
\draw[arr] (zt.east) -- (3.8, 0);

% Transition model h (decoder trapezoid: narrow left, wide right)
\fill[sprcomp, draw=black!70, thick, line join=round]
    (4.0, -0.5) -- (4.0, 0.5) -- (5.5, 0.9) -- (5.5, -0.9) -- cycle;
\node[font=\small, text=black!85] at (4.75, 0) {$h$};
\coordinate (h-north) at (4.75, 0.7);
\coordinate (h-east) at (5.5, 0);

% Action input from above
\node[font=\small, text=black!90] at (4.75, 2.0) {$a_{t+k}$};
\draw[arr] (4.75, 1.7) -- (4.75, 0.9);

% z_hat output as plain text node
\node[font=\small, text=black!90, inner sep=2pt] (zout) at (6.8, 0)
    {$\hat{z}_{t+k+1}$};
\draw[arr] (h-east) -- (zout.west);

% Feedback loop (below, back to z_t)
\draw[arr, rounded corners=12pt]
    (zout.south) -- (6.8, -1.5) -- (3.0, -1.5) -- (zt.south);

% Loop annotation
\node[lbl] at (4.9, -1.9)
    {repeat for $k = 0, 1, \ldots, K\!-\!1$};

% Final output arrow
\node[font=\small, text=black!90, inner sep=2pt] (zfinal) at (8.6, 0)
    {$\hat{z}_{t+K}$};
\draw[arr] (zout.east) -- (zfinal.west);

\end{tikzpicture}
\caption{SPR forward prediction. The encoder compresses the
  observation into a latent representation $z_t$. At each step $k$,
  the transition model $h$ takes $z_t$ and the agent's action
  $a_{t+k}$ to predict the next latent state, then feeds its output
  back as input to predict $K$ steps into the future.}
\label{fig:spr-architecture}
\end{figure}
