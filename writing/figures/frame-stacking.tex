%% Panel (a): Filmstrip of 4 frames with ball at different positions
\begin{subfigure}{\textwidth}
\centering
\begin{tikzpicture}[
    lbl/.style={font=\footnotesize, text=black!55},
    dimlbl/.style={font=\scriptsize, text=black!45},
    tlbl/.style={font=\scriptsize, text=black!50},
]
\def\c{0.28}
\def\gs{5}  % 5x5 grid per frame

% Draw one frame: #1=x-offset, #2=ball-x (grid coords), #3=ball-y, #4=opacity
\newcommand{\drawframe}[4]{
    \foreach \x in {0,...,4} {
        \foreach \y in {0,...,4} {
            \fill[black!5, draw=black!12]
                ({#1 + \x*\c}, \y*\c) rectangle ++(\c, \c);
        }
    }
    % Ball
    \fill[palA!#4] ({#1 + (#2+0.5)*\c}, {(#3+0.5)*\c}) circle (0.11);
}

% Four frames with ball moving diagonally
\drawframe{0}{1}{1}{30}        % t-3: bottom-left
\drawframe{1.7}{2}{2}{45}      % t-2: center-left
\drawframe{3.4}{3}{3}{60}      % t-1: center-right
\drawframe{5.1}{4}{4}{100}     % t:   top-right

% Time labels below each frame
\node[tlbl] at ({2.5*\c}, {-0.2}) {$t{-}3$};
\node[tlbl] at ({1.7 + 2.5*\c}, {-0.2}) {$t{-}2$};
\node[tlbl] at ({3.4 + 2.5*\c}, {-0.2}) {$t{-}1$};
\node[tlbl] at ({5.1 + 2.5*\c}, {-0.2}) {$t$};

% Arrow to stacked tensor
\pgfmathsetmacro{\arrx}{5.1 + 5*\c + 0.3}
\draw[->, thick, black!40]
    (\arrx, {2.5*\c}) -- ++(0.6, 0);

% 3D stacked tensor (4 layers with z-offset)
\pgfmathsetmacro{\tx}{\arrx + 1.15}
\def\tw{1.2}   % tensor face width
\def\th{1.2}   % tensor face height
\def\td{0.13}  % offset per layer
% Draw back to front: layer 0 (t-3) to layer 3 (t)
\foreach \l/\op in {3/15, 2/12, 1/9, 0/6} {
    \pgfmathsetmacro{\lx}{\tx + \l*\td}
    \pgfmathsetmacro{\ly}{2.5*\c - \th/2 + \l*\td}
    \fill[black!\op, draw=black!25]
        (\lx, \ly) rectangle ++({\tw}, {\th});
}
% Dimension label below the block
\node[font=\scriptsize, text=black!50, anchor=north]
    at ({\tx + \tw/2 + 1.5*\td}, {2.5*\c - \th/2 - 0.15})
    {$84 \times 84 \times 4$};

\end{tikzpicture}
\subcaption{Four consecutive frames capture the ball at different
  positions. Stacking them produces one multi-channel input.}
\label{fig:frame-stack-op}
\end{subfigure}

\vspace{1em}

%% Panel (b): Motion inference -- single frame vs stacked
\begin{subfigure}{\textwidth}
\centering
\begin{tikzpicture}[
    lbl/.style={font=\footnotesize, text=black!55},
    dimlbl/.style={font=\scriptsize, text=black!45},
]
\def\c{0.3}

% --- Single frame (left): no direction info ---
\foreach \x in {0,...,5} {
    \foreach \y in {0,...,5} {
        \fill[gray!6, draw=black!10]
            (\x*\c, \y*\c) rectangle ++(\c, \c);
    }
}
% Ball at one position only
\fill[palA] ({3.5*\c}, {3.5*\c}) circle (0.13);
\node[dimlbl] at ({3*\c}, {-0.25}) {single frame};
\node[dimlbl] at ({3*\c}, {-0.5}) {(position only)};

% --- vs label ---
\node[lbl, text=black!40] at ({6*\c + 0.5}, {3*\c}) {vs.};

% --- Stacked view (right): trajectory visible ---
\def\ox{4.3}
\foreach \x in {0,...,5} {
    \foreach \y in {0,...,5} {
        \fill[gray!6, draw=black!10]
            ({\ox + \x*\c}, \y*\c) rectangle ++(\c, \c);
    }
}

% Ball at four positions with increasing opacity
\fill[palA!25] ({\ox + 1*\c}, {1*\c}) circle (0.12);
\fill[palA!45] ({\ox + 2*\c}, {2.2*\c}) circle (0.12);
\fill[palA!65] ({\ox + 3.2*\c}, {3.2*\c}) circle (0.12);
\fill[palA]    ({\ox + 4.2*\c}, {4.5*\c}) circle (0.12);

% Trajectory arrows
\draw[->, thick, dashed, palA!35]
    ({\ox + 1*\c + 0.14}, {1*\c + 0.1})
    -- ({\ox + 2*\c - 0.1}, {2.2*\c - 0.06});
\draw[->, thick, dashed, palA!50]
    ({\ox + 2*\c + 0.14}, {2.2*\c + 0.1})
    -- ({\ox + 3.2*\c - 0.1}, {3.2*\c - 0.06});
\draw[->, thick, dashed, palA!65]
    ({\ox + 3.2*\c + 0.14}, {3.2*\c + 0.1})
    -- ({\ox + 4.2*\c - 0.1}, {4.5*\c - 0.06});

\node[dimlbl] at ({\ox + 3*\c}, {-0.25}) {four stacked frames};
\node[dimlbl] at ({\ox + 3*\c}, {-0.5}) {(position + direction)};

\end{tikzpicture}
\subcaption{A single frame shows where the ball is but not where
  it is going. Stacking four frames reveals the trajectory.}
\label{fig:frame-stack-motion}
\end{subfigure}

\caption{Frame stacking provides motion information.
  (a)~Each frame captures a snapshot; stacking concatenates them.
  (b)~The network reads velocity from positions across frames.}
\label{fig:frame-stacking}
