% Encoder colors (reuse from Nature CNN)
\definecolor{encfade}{HTML}{B8D2E6}
% SPR-specific component color
\definecolor{sprcomp}{HTML}{C2D4A8}
% Target path (faded)
\definecolor{targetfade}{HTML}{E0E0E0}

%% --- (a) Forward prediction ---
\begin{subfigure}[t]{\textwidth}
\centering
\begin{tikzpicture}[
    arr/.style={->, thick, >=stealth, black!55},
    lbl/.style={font=\scriptsize, text=black!55},
]

% s_t
\node[font=\small, text=black!90] at (-0.5, 0) {$s_t$};
\draw[arr] (-0.1, 0) -- (0.6, 0);

% Encoder (trapezoid: wide left, narrow right = compression)
\fill[encfade, draw=black!70, thick, line join=round]
    (0.8, -0.8) -- (0.8, 0.8) -- (2.1, 0.45) -- (2.1, -0.45) -- cycle;
\node[font=\small, text=black!85] at (1.35, 0) {$f_\theta$};

% z_t as plain text node -- arrows point to and from it
\node[font=\small, text=black!90, inner sep=2pt] (zt) at (3.0, 0) {$z_t$};
\draw[arr] (2.1, 0) -- (zt.west);
\draw[arr] (zt.east) -- (3.8, 0);

% Transition model h (decoder trapezoid: narrow left, wide right)
\fill[sprcomp, draw=black!70, thick, line join=round]
    (4.0, -0.5) -- (4.0, 0.5) -- (5.5, 0.9) -- (5.5, -0.9) -- cycle;
\node[font=\small, text=black!85] at (4.75, 0) {$h$};
\coordinate (h-north) at (4.75, 0.7);
\coordinate (h-east) at (5.5, 0);

% Action input from above
\node[font=\small, text=black!90] at (4.75, 2.0) {$a_{t+k}$};
\draw[arr] (4.75, 1.7) -- (h-north);

% z_hat output as plain text node
\node[font=\small, text=black!90, inner sep=2pt] (zout) at (6.8, 0)
    {$\hat{z}_{t+k+1}$};
\draw[arr] (h-east) -- (zout.west);

% Feedback loop (below, back to z_t)
\draw[arr, rounded corners=12pt]
    (zout.south) -- (6.8, -1.5) -- (3.0, -1.5) -- (zt.south);

% Loop annotation
\node[lbl] at (4.9, -1.9)
    {repeat for $k = 0, 1, \ldots, K\!-\!1$};

% Final output arrow
\node[font=\small, text=black!90, inner sep=2pt] (zfinal) at (8.6, 0)
    {$\hat{z}_{t+K}$};
\draw[arr] (zout.east) -- (zfinal.west);

\end{tikzpicture}
\caption{The encoder compresses the observation into a latent
  representation $z_t$. The transition model $h$ predicts the
  next latent state from $z_t$ and the agent's action, then feeds
  its output back as input to predict $K$ steps into the future.}
\label{fig:spr-forward}
\end{subfigure}

%% --- (b) Self-predictive loss ---
\begin{subfigure}[t]{\textwidth}
\centering
\begin{tikzpicture}[
    box/.style={draw=black!70, thick, minimum height=0.7cm,
                font=\small, inner sep=5pt, rounded corners=2pt},
    arr/.style={->, thick, >=stealth, black!55},

    mathlbl/.style={font=\small, text=black!65},
    pathlbl/.style={font=\small\bfseries, text=black!55},
]

% === Online path (top) ===
\node[font=\small\bfseries, text=black!80] at (-0.5, 1.6) {Online};

% z_hat_{t+k} input
\node[font=\small, text=black!90] at (-0.3, 0) {$\hat{z}_{t\!+\!k}$};
\draw[arr] (0.3, 0) -- (1.0, 0);

% Projection
\node[box, fill=sprcomp] (proj) at (1.8, 0) {$g$};
\node[lbl] at (1.8, -0.75) {project};

\draw[arr] (proj.east) -- (3.2, 0);

% Prediction head
\node[box, fill=sprcomp] (pred) at (4.0, 0) {$q$};
\node[lbl] at (4.0, -0.75) {predict};

% Online output
\draw[arr] (pred.east) -- (5.5, 0);
\node[font=\small, text=black!90] at (5.85, 0) {$\tilde{y}$};

% s_{t+k}

% EMA Encoder (trapezoid)
\fill[targetfade, draw=black!55, thick, line join=round]
    (0.8, -4.4) -- (0.8, -2.8) -- (2.1, -3.15) -- (2.1, -4.05) -- cycle;
\node[font=\small, text=black!70] at (1.35, -3.6) {$\bar{f}_\theta$};

\draw[arr, black!50] (2.1, -3.6) -- (2.8, -3.6);

% EMA Projection
\node[box, fill=targetfade, draw=black!55, text=black!70]
    (tproj) at (3.6, -3.6) {$\bar{g}$};

% Target output
\draw[arr, black!50] (tproj.east) -- (5.1, -3.6);
\node[font=\small, text=black!70] at (5.45, -3.6) {$\bar{y}$};

% === Cosine similarity ===
\draw[<->, thick, black!75]
    (5.85, -0.4) -- (5.85, -1.5);
\draw[->, thick, dashed, black!55]
    (5.45, -3.2) -- (5.85, -1.9);
\node[font=\small, text=black!85, anchor=west] at (6.2, -1.0)
    {$\mathcal{L}_{\text{SPR}} = -\cos(\tilde{y},\, \bar{y})$};

% No prediction head annotation
\draw[decorate, decoration={brace, amplitude=3pt, mirror}, black!45]
    (2.5, -4.5) -- (4.2, -4.5)
    node[midway, below=4pt, font=\scriptsize, text=black!50]
    {no prediction head (asymmetry)};

% EMA labels (under encoder and projection)
\node[font=\scriptsize, text=black!50] at (1.45, -4.65) {EMA};
\node[font=\scriptsize, text=black!50] at (3.6, -4.25) {EMA};

% EMA update arrow -- label offset left to avoid overlap
\node[font=\scriptsize, text=black!50, align=center, anchor=east]
    at (-0.75, -1.4) {EMA\\update};
\draw[->, thick, dashed, black!45]
    (-0.5, -0.4) -- (-0.5, -2.4);

\end{tikzpicture}
\caption{Online predictions are compared to EMA targets via cosine
  similarity. The prediction head $q$ exists only on the online
  side, preventing representation collapse.}
\label{fig:spr-loss}
\end{subfigure}

\caption{SPR training architecture. (a)~The transition model
  predicts future representations from the current state.
  (b)~Predictions are trained against targets from an EMA copy
  of the encoder. Gradients flow through the online path only.}
\label{fig:spr-architecture}
\end{figure}
