%% Panel (a): The stacking operation
\begin{subfigure}[t]{0.42\textwidth}
\centering
\begin{tikzpicture}[
    lbl/.style={font=\scriptsize, text=black!55},
    dimlbl/.style={font=\tiny, text=black!45},
]
\def\c{0.28}
\def\s{5}  % grid size

% Four stacked frames (back to front)
\foreach \d/\shade/\tlbl in {3/12/$t{-}3$, 2/10/$t{-}2$, 1/8/$t{-}1$, 0/5/$t$} {
    \pgfmathsetmacro{\dx}{\d * 0.09}
    \pgfmathsetmacro{\dy}{\d * 0.09}
    \foreach \x in {0,...,4} {
        \foreach \y in {0,...,4} {
            \fill[black!\shade, draw=black!12]
                ({\dx + \x*\c}, {\dy + \y*\c})
                rectangle ++({\c}, {\c});
        }
    }
}

% Frame labels on right side
\node[dimlbl, anchor=west] at ({5*\c + 0.35}, {2.5*\c + 0.27}) {$t{-}3$};
\node[dimlbl, anchor=west] at ({5*\c + 0.35}, {2.5*\c + 0.09}) {$t{-}1$};
\node[dimlbl, anchor=west] at ({5*\c + 0.35}, {2.5*\c}) {$t$};

% Arrow
\draw[->, thick, black!40]
    ({5*\c + 0.9}, {2.5*\c + 0.14}) -- ++(0.6, 0);

% Output block
\node[draw=black!30, thick, rounded corners=2pt,
      minimum width=1.1cm, minimum height=1.3cm,
      fill=black!6, font=\scriptsize, align=center]
    (out) at ({5*\c + 2.15}, {2.5*\c + 0.14})
    {$84{\times}84$\\${\times}\,4$};

\end{tikzpicture}
\subcaption{Four consecutive frames are concatenated along the
  channel dimension.}
\label{fig:frame-stack-op}

%% Panel (b): Motion inference
\begin{subfigure}[t]{0.52\textwidth}
\centering
\begin{tikzpicture}[
    lbl/.style={font=\scriptsize, text=black!55},
    dimlbl/.style={font=\tiny, text=black!45},
]
\def\c{0.3}

% Game-like grid (8x6)
\foreach \x in {0,...,7} {
    \foreach \y in {0,...,5} {
        \fill[gray!6, draw=black!10]
            (\x*\c, \y*\c) rectangle ++(\c, \c);
    }
}

% Ball at three positions (showing diagonal motion)
% t-2: bottom-left area
\fill[palA!30] ({1.5*\c}, {1.5*\c}) circle (0.12);
\node[dimlbl, anchor=north] at ({1.5*\c}, {1.5*\c - 0.15})
    {$t{-}2$};

% t-1: center area
\fill[palA!50] ({3.5*\c}, {3*\c}) circle (0.12);
\node[dimlbl, anchor=north] at ({3.5*\c}, {3*\c - 0.15})
    {$t{-}1$};

% t: upper-right area
\fill[palA] ({5.5*\c}, {4.5*\c}) circle (0.12);
\node[dimlbl, anchor=north] at ({5.5*\c}, {4.5*\c - 0.15})
    {$t$};

% Dashed trajectory arrow
\draw[->, thick, dashed, palA!40]
    ({1.5*\c + 0.15}, {1.5*\c + 0.1})
    -- ({3.5*\c - 0.12}, {3*\c - 0.06});
\draw[->, thick, dashed, palA!50]
    ({3.5*\c + 0.15}, {3*\c + 0.1})
    -- ({5.5*\c - 0.12}, {4.5*\c - 0.06});

% Annotation
\node[lbl, anchor=north, align=center] at ({4*\c}, -0.15)
    {direction inferred from\\positions across frames};

\end{tikzpicture}
\subcaption{A ball's positions across three frames reveal its
  trajectory -- information unavailable from any single frame.}
\label{fig:frame-stack-motion}
\end{subfigure}
