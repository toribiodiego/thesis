\begin{figure}[ht]
\centering
\begin{tikzpicture}[
    lbl/.style={font=\footnotesize, text=black!55},
    dimlbl/.style={font=\scriptsize, text=black!45},
]
% Same cell size for both grids -- size difference shows resolution
\def\c{0.15}

% --- Shared face pattern evaluator ---
% Evaluates the 21x16 face at source coordinates (continuous)
% Sets \facetype to 0=bg, 1=face, 2=outline, 3=feature
% (Used by both grids; the small grid maps its cells to source coords)

% --- Original: 21x16 grid ---
\foreach \x in {0,...,20} {
    \foreach \y in {0,...,15} {
        \pgfmathsetmacro{\dist}{sqrt((\x-10)*(\x-10)+(\y-7.5)*(\y-7.5))}
        \pgfmathtruncatemacro{\inface}{\dist < 6.3 ? 1 : 0}
        \pgfmathtruncatemacro{\onedge}{(\dist >= 6.3)*(\dist < 7.2) ? 1 : 0}
        \pgfmathtruncatemacro{\feat}{(
            (\x>=7)*(\x<=8)*(\y>=9)*(\y<=10) +
            (\x>=12)*(\x<=13)*(\y>=9)*(\y<=10) +
            (\x==6)*(\y==5) + (\x==14)*(\y==5) +
            (\x==7)*(\y==4) + (\x==13)*(\y==4) +
            (\x>=8)*(\x<=12)*(\y==3)
        ) > 0 ? 1 : 0}
        \ifnum\onedge=1
            \fill[black!55, draw=black!5]
                (\x*\c, \y*\c) rectangle ++(\c, \c);
        \else\ifnum\inface=0
            \fill[black!6, draw=black!5]
                (\x*\c, \y*\c) rectangle ++(\c, \c);
        \else\ifnum\feat=1
            \fill[black!85, draw=black!5]
                (\x*\c, \y*\c) rectangle ++(\c, \c);
        \else
            \fill[black!20, draw=black!5]
                (\x*\c, \y*\c) rectangle ++(\c, \c);
        \fi\fi\fi
    }
}
\node[lbl] at ({10.5*\c}, {16*\c + 0.5}) {original};
\node[dimlbl, anchor=north] at ({10.5*\c}, -0.08)
    {$210 \times 160$};

% --- Arrow ---
\draw[->, thick, black!35]
    ({21*\c + 0.3}, {8*\c}) -- ++(0.5, 0);

% --- Downsampled: 8x8 grid, same cell size ---
% Nearest-neighbor: map each cell back to 21x16 source coords
% Vertically center the 8x8 grid with the 21x16 grid
\pgfmathsetmacro{\ox}{21*\c + 1.1}
\pgfmathsetmacro{\oy}{(16*\c - 8*\c) / 2}
\foreach \x in {0,...,7} {
    \foreach \y in {0,...,7} {
        % Map to source coordinates in 21x16 space
        \pgfmathsetmacro{\xsrc}{(\x + 0.5) * 21.0 / 8.0 - 0.5}
        \pgfmathsetmacro{\ysrc}{(\y + 0.5) * 16.0 / 8.0 - 0.5}
        % Evaluate same face pattern at source coords
        \pgfmathsetmacro{\dist}{sqrt((\xsrc-10)*(\xsrc-10)+(\ysrc-7.5)*(\ysrc-7.5))}
        \pgfmathtruncatemacro{\inface}{\dist < 6.3 ? 1 : 0}
        \pgfmathtruncatemacro{\onedge}{(\dist >= 6.3)*(\dist < 7.2) ? 1 : 0}
        \pgfmathtruncatemacro{\xr}{round(\xsrc)}
        \pgfmathtruncatemacro{\yr}{round(\ysrc)}
        \pgfmathtruncatemacro{\feat}{(
            (\xr>=7)*(\xr<=8)*(\yr>=9)*(\yr<=10) +
            (\xr>=12)*(\xr<=13)*(\yr>=9)*(\yr<=10) +
            (\xr==6)*(\yr==5) + (\xr==14)*(\yr==5) +
            (\xr==7)*(\yr==4) + (\xr==13)*(\yr==4) +
            (\xr>=8)*(\xr<=12)*(\yr==3)
        ) > 0 ? 1 : 0}
        \ifnum\onedge=1
            \fill[black!55, draw=black!8]
                ({\ox + \x*\c}, {\oy + \y*\c}) rectangle ++(\c, \c);
        \else\ifnum\inface=0
            \fill[black!6, draw=black!8]
                ({\ox + \x*\c}, {\oy + \y*\c}) rectangle ++(\c, \c);
        \else\ifnum\feat=1
            \fill[black!85, draw=black!8]
                ({\ox + \x*\c}, {\oy + \y*\c}) rectangle ++(\c, \c);
        \else
            \fill[black!20, draw=black!8]
                ({\ox + \x*\c}, {\oy + \y*\c}) rectangle ++(\c, \c);
        \fi\fi\fi
    }
}
\node[lbl] at ({\ox + 4*\c}, {16*\c + 0.5}) {downsampled};
\node[dimlbl, anchor=north] at ({\ox + 4*\c}, {\oy - 0.08})
    {$84 \times 84$};

\end{tikzpicture}
\caption{Downsampling. The frame is resized from $210 \times 160$
  to $84 \times 84$. The face is still recognizable at lower
  resolution.}
\label{fig:resize}
\end{figure}
