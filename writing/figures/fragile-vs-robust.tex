\newcommand{\ponggrid}[3]{%
    % #1=x-off, #2=y-off, #3=cell shade (for dimming)
    \def\c{0.38}
    \foreach \x in {0,...,7} {
        \foreach \y in {0,...,5} {
            \fill[black!#3, draw=black!8]
                ({#1 + \x*\c}, {#2 + \y*\c}) rectangle ++(\c, \c);
        }
    }
}

%% (a) Fragile, original -- spotlight on ball
\begin{subfigure}[t]{0.45\textwidth}
\centering
\begin{tikzpicture}
    \def\c{0.38}
    % Dimmed grid
    \ponggrid{0}{0}{6}
    % Bright spotlight region (3x3 cells around position 4,4)
    \foreach \x in {3,...,5} {
        \foreach \y in {3,...,5} {
            \fill[white, draw=black!8]
                (\x*\c, \y*\c) rectangle ++(\c, \c);
        }
    }
    \draw[palB, very thick, rounded corners=2pt]
        ({3*\c}, {3*\c}) rectangle ({6*\c}, {6*\c});
    % Paddle (dimmed)
    \fill[black!35, rounded corners=1pt]
        ({7*\c + 0.03}, {1.5*\c})
        rectangle ({8*\c - 0.03}, {4.5*\c});
    % Ball INSIDE spotlight
    \fill[palA] ({4.5*\c}, {4.5*\c}) circle (0.13);
    % Spotlight label
    \node[font=\scriptsize, text=palB!80, anchor=south]
        at ({4.5*\c}, {6*\c + 0.08}) {encoder looks here};
    % Verdict
    \node[font=\footnotesize, text=palC!80]
        at ({4*\c}, {-0.35}) {ball found -- correct};
\end{tikzpicture}
\subcaption{Original: the ball is inside the fixed region.}
\label{fig:fragile-original}
\end{subfigure}
\hfill

%% (c) Robust, original -- tracks relationship
\begin{subfigure}[t]{0.45\textwidth}
\centering
\begin{tikzpicture}
    \def\c{0.38}
    % Normal grid (not dimmed -- encoder sees everything)
    \ponggrid{0}{0}{3}
    % Paddle
    \fill[black!45, rounded corners=1pt]
        ({7*\c + 0.03}, {1.5*\c})
        rectangle ({8*\c - 0.03}, {4.5*\c});
    % Ball
    \fill[palA] ({4.5*\c}, {4.5*\c}) circle (0.13);
    % Green highlights around objects
    \draw[palC, very thick, rounded corners=3pt]
        ({4*\c}, {4*\c}) rectangle ({5*\c}, {5*\c});
    \draw[palC, very thick, rounded corners=3pt]
        ({7*\c - 0.04}, {1.3*\c})
        rectangle ({8*\c + 0.04}, {4.7*\c});
    % Relationship
    \draw[palC, very thick, <->]
        ({5*\c + 0.06}, {4.3*\c}) -- ({7*\c - 0.08}, {3.2*\c});
    \node[font=\scriptsize, text=palC!70, anchor=south]
        at ({6*\c}, {4.0*\c}) {distance};
    % Verdict
    \node[font=\footnotesize, text=palC!80]
        at ({4*\c}, {-0.35}) {relationship found -- correct};
\end{tikzpicture}
\subcaption{Original: encoder tracks ball-paddle distance.}
\label{fig:robust-original}
\end{subfigure}
