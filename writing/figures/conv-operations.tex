%% Panel (a): Computing one output value
\begin{subfigure}{\textwidth}
\centering
\begin{tikzpicture}[
    lbl/.style={font=\scriptsize, text=black!55},
    op/.style={font=\normalsize, text=black!45},
]
\def\c{0.6}

% --- Input patch (3x3) ---
\foreach \x/\y/\v in {
    0/2/2, 1/2/0, 2/2/1,
    0/1/1, 1/1/1, 2/1/0,
    0/0/0, 1/0/1, 2/0/2}
{
    \fill[palA!10, draw=palA!30]
        (\x*\c, \y*\c) rectangle ++(\c, \c);
    \node[font=\small] at ({(\x+0.5)*\c}, {(\y+0.5)*\c}) {\v};
}
\node[lbl, anchor=north] at ({1.5*\c}, -0.12) {input patch};

% --- Element-wise multiply symbol ---
\node[op] at ({3*\c + 0.35}, {1.5*\c}) {$\odot$};

% --- Kernel (3x3) ---
\def\kx{2.5}
\foreach \x/\y/\v in {
    0/2/1, 1/2/0, 2/2/1,
    0/1/0, 1/1/1, 2/1/0,
    0/0/1, 1/0/0, 2/0/1}
{
    \fill[palB!6, draw=palB!22]
        ({\kx + \x*\c}, \y*\c) rectangle ++({\c}, {\c});
    \node[font=\small] at ({\kx + (\x+0.5)*\c}, {(\y+0.5)*\c}) {\v};
}
\node[lbl, anchor=north] at ({\kx + 1.5*\c}, -0.12) {kernel};

% --- Equals ---
\node[op] at ({\kx + 3*\c + 0.35}, {1.5*\c}) {$=$};

% --- Output value ---
\node[fill=palA!15, draw=palA!45, rounded corners=2pt,
      inner sep=5pt, font=\bfseries]
    at ({\kx + 3*\c + 1.0}, {1.5*\c}) {$6$};

% --- Computation string (centered under full panel) ---
\node[font=\scriptsize, text=black!50, anchor=north]
    at ({(\kx + 3*\c + 1.0) / 2}, {-0.5})
    {$2{\cdot}1 + 0{\cdot}0 + 1{\cdot}1 + 1{\cdot}0
      + 1{\cdot}1 + 0{\cdot}0 + 0{\cdot}1 + 1{\cdot}0
      + 2{\cdot}1 = 6$};

\end{tikzpicture}
\subcaption{At each position, corresponding input and kernel values
  are multiplied and summed into one output value.}
\label{fig:conv-op-compute}
\end{subfigure}

\vspace{1.2em}

%% Panel (b): Sliding across the input
\begin{subfigure}{\textwidth}
\centering
\begin{tikzpicture}[
    arr/.style={->, thick, >=stealth, black!40},
    lbl/.style={font=\scriptsize, text=black!55},
]
\def\c{0.55}

% --- Full input grid (5x5) ---
% Row 0 at top (y=4), row 4 at bottom (y=0)
\foreach \col/\row/\v in {
    0/4/2, 1/4/0, 2/4/1, 3/4/0, 4/4/3,
    0/3/1, 1/3/1, 2/3/0, 3/3/2, 4/3/0,
    0/2/0, 1/2/1, 2/2/2, 3/2/0, 4/2/1,
    0/1/1, 1/1/0, 2/1/0, 3/1/1, 4/1/0,
    0/0/0, 1/0/2, 2/0/1, 3/0/0, 4/0/1}
{
    \fill[gray!6, draw=black!15]
        (\col*\c, \row*\c) rectangle ++(\c, \c);
    \node[font=\scriptsize]
        at ({(\col+0.5)*\c}, {(\row+0.5)*\c}) {\v};
}

% Position 1 border (solid blue, rows 0-2, cols 0-2)
\draw[palA, very thick]
    (0, {2*\c}) rectangle ({3*\c}, {5*\c});

% Position 2 border (dashed green, rows 0-2, cols 2-4)
\draw[palC, thick, dashed]
    ({2*\c}, {2*\c}) rectangle ({5*\c}, {5*\c});

% Kernel size brace on LEFT side (avoids overlap with position 2)
\draw[decorate, decoration={brace, amplitude=3pt, mirror}, black!40]
    (-0.1, {5*\c}) -- (-0.1, {2*\c})
    node[midway, left=4pt, font=\scriptsize, text=black!45]
    {$3{\times}3$};

% Stride arrow above grid (from pos 1 center to pos 2 center)
\draw[->, thick, black!45]
    ({1.5*\c}, {5*\c + 0.25}) -- ({3.5*\c}, {5*\c + 0.25})
    node[midway, above, font=\scriptsize, text=black!50]
    {stride $2$};

% Input label
\node[lbl, anchor=north] at ({2.5*\c}, -0.15) {input};

% --- Arrow ---
\draw[arr] ({5*\c + 0.55}, {2.5*\c}) -- ++(1.1, 0);

% --- Output grid (2x2) ---
% Vertically centered with input grid
\def\ox{5.0}
\def\oy{0.825}

% Top-left: value 6, blue fill (matches position 1)
\fill[palA!12, draw=palA!50, thick]
    (\ox, {\oy + \c}) rectangle ++(\c, \c);
\node[font=\scriptsize] at ({\ox + 0.5*\c}, {\oy + 1.5*\c}) {6};

% Top-right: value 9, green fill (matches position 2)
\fill[palC!10, draw=palC!40]
    ({\ox + \c}, {\oy + \c}) rectangle ++(\c, \c);
\node[font=\scriptsize] at ({\ox + 1.5*\c}, {\oy + 1.5*\c}) {9};

% Bottom-left: value 3
\fill[gray!5, draw=black!15]
    (\ox, \oy) rectangle ++(\c, \c);
\node[font=\scriptsize] at ({\ox + 0.5*\c}, {\oy + 0.5*\c}) {3};

% Bottom-right: value 6
\fill[gray!5, draw=black!15]
    ({\ox + \c}, \oy) rectangle ++(\c, \c);
\node[font=\scriptsize] at ({\ox + 1.5*\c}, {\oy + 0.5*\c}) {6};

% Output label
\node[lbl, anchor=north] at ({\ox + \c}, {\oy - 0.15}) {output};

\end{tikzpicture}
\subcaption{The kernel slides with stride~$2$, producing one value
  per position. Colors match input positions to output cells.
  Output size: $\lfloor(5{-}3)/2\rfloor + 1 = 2$.}
\label{fig:conv-op-slide}
\end{subfigure}

\caption{Convolution operation. (a)~Multiply-and-sum at one position.
  (b)~Sliding the kernel produces the output feature map.}
\label{fig:conv-operation}
\end{figure}
