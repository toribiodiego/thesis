%% Right: same path as a state-transition graph
\begin{subfigure}[b]{0.52\textwidth}
\centering
\begin{tikzpicture}[
    state/.style={circle, draw=black!50, thick,
                  minimum size=1.1cm, font=\small},
    active/.style={->, thick, >=stealth, agentblue!65,
                   shorten >=2pt, shorten <=2pt},
    faded/.style={->, thick, >=stealth, black!18,
                  shorten >=2pt, shorten <=2pt},
    alabel/.style={font=\footnotesize, text=black!65},
    rlabel/.style={font=\footnotesize, text=black!45},
    fadea/.style={font=\footnotesize, text=black!22},
]

% States
\node[state] (s01) at (0, 0)     {$s_{01}$};
\node[state] (s02) at (0, 2.4)   {$s_{02}$};
\node[state] (s12) at (3.2, 2.4) {$s_{12}$};
\node[state] (s11) at (3.2, 0)   {$s_{11}$};

% Labels
\node[font=\scriptsize, text=black!45,
      above=2pt] at (s12.north) {toward goal};
\node[font=\scriptsize, text=penaltyred,
      below=2pt] at (s11.south) {penalty};

% Active path: s01 -> s02 (up)
\draw[active] (s01) to[bend left=15]
    node[alabel, left] {up}
    node[rlabel, left, yshift=-10pt] {$r\!=\!0$}
    (s02);

% Active path: s02 -> s12 (right)
\draw[active] (s02) to[bend left=10]
    node[alabel, above] {right}
    node[rlabel, below] {$r\!=\!0$}
    (s12);

% Faded: s01 -> s11 (right, toward penalty)
\draw[faded] (s01) to[bend right=10]
    node[fadea, below] {right}
    (s11);

% Faded: s02 -> s01 (down, back)
\draw[faded] (s02) to[bend left=15]
    node[fadea, right] {down}
    (s01);

\end{tikzpicture}

\subcaption{The same transitions as a directed graph.}
