% Object A: dark circle (shifted position)

\fill[black!55, rounded corners=2pt]
    (2.2*\c, 0.5*\c) rectangle (2.8*\c, 2.0*\c);
\node[font=\scriptsize, text=black!50, anchor=west]
    at (2.9*\c, 1.25*\c) {B};

% Blue outlines tracking objects
\draw[palA, very thick, rounded corners=4pt]
    (4.9*\c, 2.9*\c) rectangle (6.1*\c, 4.2*\c);

    (2.0*\c, 0.3*\c) rectangle (3.0*\c, 2.2*\c);

% Distance line

    (4.9*\c, 2.9*\c) -- (3.0*\c, 2.2*\c);

      fill=white, inner sep=2pt, rounded corners=1pt]
    at (3.95*\c, 2.8*\c) {$d$};

\subcaption{With augmentation: the encoder tracks the distance
  between objects regardless of their position.}
\label{fig:robust-feature}
\end{subfigure}
