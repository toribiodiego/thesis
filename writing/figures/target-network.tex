\node[font=\scriptsize, text=black!40] at (2.2, -1.3)
    {frozen};

    node[lbl, midway, above, text=black!50]
    {copy every $C$ steps};

% Prediction label
\node[lbl, below=1pt] at (-2.2, -2.95)
    {prediction};

    {target};

\end{tikzpicture}
