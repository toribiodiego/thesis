%% (b) Robust: tracks object relationship
\begin{subfigure}[t]{0.47\textwidth}
\centering
\begin{tikzpicture}

% Plain background
\fill[black!4] (0, 0) rectangle (5, 4);
\draw[black!15] (0, 0) rectangle (5, 4);

% Object A: circle (different position -- shifted)
\fill[palA] (1.5, 3.0) circle (0.25);
\node[font=\scriptsize, text=palA!70, anchor=south]
    at (1.5, 3.35) {A};

% Object B: rounded square (also shifted)
\fill[palA!60, rounded corners=2pt]
    (3.5, 0.6) rectangle (4.1, 1.6);
\node[font=\scriptsize, text=palA!50, anchor=south]
    at (3.8, 1.7) {B};

% Green outlines tracking both objects
\draw[palC, very thick, rounded corners=4pt]
    (1.1, 2.55) rectangle (1.9, 3.5);
\draw[palC, very thick, rounded corners=4pt]
    (3.35, 0.45) rectangle (4.25, 1.8);

% Distance line between objects
\draw[palC, thick, dashed]
    (1.7, 2.55) -- (3.55, 1.8);
\node[font=\footnotesize, text=palC!80,
      fill=white, inner sep=2pt, rounded corners=1pt]
    at (2.6, 2.1) {$d$};

\end{tikzpicture}
\subcaption{With augmentation: the encoder tracks the distance
  between objects -- a feature that survives spatial shifts.}
\label{fig:robust-feature}
\end{subfigure}

\caption{Augmentation as regularization. (a)~A position-based
  feature works only when objects are at memorized coordinates.
  (b)~A relationship-based feature works regardless of position.}
\label{fig:fragile-vs-robust}
