\newcommand{\pongscene}[4]{%
    % #1=x-off, #2=y-off, #3=ball-col (grid), #4=ball-row (grid)
    \def\c{0.38}
    \foreach \x in {0,...,7} {
        \foreach \y in {0,...,5} {
            \fill[black!3, draw=black!8]
                ({#1 + \x*\c}, {#2 + \y*\c}) rectangle ++(\c, \c);
        }
    }
    % Paddle
    \fill[black!45, rounded corners=1pt]
        ({#1 + 7*\c + 0.03}, {#2 + 1.5*\c})
        rectangle ({#1 + 8*\c - 0.03}, {#2 + 4.5*\c});
    % Ball
    \fill[palA] ({#1 + (#3+0.5)*\c}, {#2 + (#4+0.5)*\c}) circle (0.12);
}

%% (a) Fragile, original -- crosshair hits ball
\begin{subfigure}[t]{0.45\textwidth}
\centering
\begin{tikzpicture}
    \pongscene{0}{0}{4}{4}
    \def\c{0.38}
    % Red crosshair at fixed position (4,4)
    \draw[palB, very thick, dashed]
        ({4*\c + 0.04}, {4*\c + 0.04})
        rectangle ({5*\c - 0.04}, {5*\c - 0.04});
    \draw[palB, very thick]
        ({4.5*\c - 0.2}, {4.5*\c}) -- ({4.5*\c + 0.2}, {4.5*\c});
    \draw[palB, very thick]
        ({4.5*\c}, {4.5*\c - 0.2}) -- ({4.5*\c}, {4.5*\c + 0.2});
    % Label with arrow pointing to crosshair
    \draw[palB!70, thick, ->]
        ({6.2*\c}, {5.5*\c}) -- ({5.1*\c}, {4.8*\c});
    \node[font=\scriptsize, text=palB!70, anchor=south west]
        at ({5.8*\c}, {5.5*\c}) {fixed pixel};
\end{tikzpicture}
