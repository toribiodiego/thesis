\begin{figure}[htbp]
\centering
\input{figures/palette}
\definecolor{encfade}{HTML}{B8D2E6}
\definecolor{sprcomp}{HTML}{C2D4A8}

\begin{tikzpicture}[
    arr/.style={->, thick, >=stealth, black!55},
    gradarr/.style={->, thick, >=stealth, dashed},
    box/.style={draw=black!70, thick, minimum height=0.8cm,
                minimum width=1.6cm,
                font=\small, inner sep=5pt, rounded corners=2pt},
    lbl/.style={font=\scriptsize, text=black!55},
]

% === Observation input ===
\node[font=\small, text=black!90] (obs) at (-0.8, 0) {$s_t$};
\draw[arr] (-0.4, 0) -- (0.5, 0);

% === Shared encoder (trapezoid) ===
\fill[encfade, draw=black!70, thick, line join=round]
    (0.7, -1.1) -- (0.7, 1.1) -- (2.4, 0.6) -- (2.4, -0.6) -- cycle;
\node[font=\small, text=black!85] at (1.45, 0) {$f_\theta$};

% === Bottleneck representation ===
\node[font=\small, text=black!90, inner sep=2pt] (z) at (3.3, 0)
    {$\mathbf{z}_t$};
\draw[arr] (2.4, 0) -- (z.west);

% === Q-learning path (upper) ===
\draw[arr, palA] (z.north) -- ++(0, 0.75) coordinate (qfork);
\draw[arr, palA] (qfork) -- ++(2.3, 0);
\node[box, fill=palA!12] (qhead) at (7.1, 1.5)
    {Q-head $\;\to\; \mathcal{L}_{\text{Q}}$};
\node[lbl, text=palA!70!black, anchor=north] at (7.1, 0.95)
    {action values};

% === SSL path (lower) ===
\draw[arr, palC!80!black] (z.south) -- ++(0, -0.75) coordinate (sslfork);
\draw[arr, palC!80!black] (sslfork) -- ++(2.3, 0);
\node[box, fill=sprcomp!70] (sslhead) at (7.1, -1.5)
    {SSL head $\;\to\; \lambda\,\mathcal{L}_{\text{SSL}}$};
\node[lbl, text=palC!60!black, anchor=north] at (7.1, -2.05)
    {dynamics / structure};

% === Gradient arrows (routed outside) ===
% Q gradient (blue, dashed, above)
\draw[gradarr, palA!65, rounded corners=8pt]
    (qhead.west |- 0, 2.3) -- (1.45, 2.3) -- (1.45, 1.15);
\draw[thick, dashed, palA!65] (qhead.north) -- ++(0, 0.23);
\node[font=\scriptsize, text=palA!65] at (4.3, 2.6)
    {$\nabla \mathcal{L}_{\text{Q}}$};

% SSL gradient (green, dashed, below)
\draw[gradarr, palC!55!black, rounded corners=8pt]
    (sslhead.west |- 0, -2.3) -- (1.45, -2.3) -- (1.45, -1.15);
\draw[thick, dashed, palC!55!black] (sslhead.south) -- ++(0, -0.22);
\node[font=\scriptsize, text=palC!55!black] at (4.3, -2.6)
    {$\nabla \lambda\mathcal{L}_{\text{SSL}}$};

% === Labels ===
\node[lbl, anchor=north] at (1.45, -1.4) {shared encoder};
\node[lbl, anchor=north] at (3.3, -0.35) {\footnotesize 512-dim};

\end{tikzpicture}
\caption{Dual-objective training. The shared encoder $f_\theta$
  produces a representation $\mathbf{z}_t$ that feeds both the
  Q-head and the self-supervised head. Gradients from both losses
  flow back through the encoder, shaping its representations with
  complementary signals.}
\label{fig:dual-objective}
\end{figure}
