\centering
\input{figures/palette}
\definecolor{nodeblue}{HTML}{B8D2E6}
\definecolor{nodeborder}{HTML}{5A87B5}
\begin{tikzpicture}[
    val/.style={font=\small, text=black!60, inner sep=2pt},
    opnode/.style={rectangle, draw=nodeborder, thick, fill=nodeblue!30,
                    minimum width=28pt, minimum height=28pt,
                    inner sep=2pt, rounded corners=2pt},
    actnode/.style={rectangle, draw=nodeborder, thick, fill=nodeblue!30,
                    minimum width=28pt, minimum height=28pt,
                    inner sep=2pt, rounded corners=2pt},
    conn/.style={->, thick, >=stealth, black!45},
    mathlbl/.style={font=\small, text=black!70},
    seclbl/.style={font=\small, text=black!55},
]

% --- Input values ---
\node[val] (x1) at (-0.6, 1.5)  {$x_1$};
\node[val] (x2) at (-0.6, 0)    {$x_2$};
\node[val] (x3) at (-0.6, -1.5) {$x_3$};

% --- Summation node ---
\node[opnode] (sum) at (2.8, 0) {\small $\Sigma$};

% --- Connections: input to summation, with weight labels ---
\draw[conn, shorten >=4pt, shorten <=6pt] (x1) -- (sum) node[mathlbl, midway, above, sloped] {$w_1$};
\draw[conn, shorten >=4pt, shorten <=6pt] (x2) -- (sum) node[mathlbl, midway, above, sloped] {$w_2$};
\draw[conn, shorten >=4pt, shorten <=6pt] (x3) -- (sum) node[mathlbl, midway, above, sloped] {$w_3$};

% --- Bias addition (separate from weighted sum) ---
\node[val, font=\Large, text=black!55] (add) at (4.6, 0) {$\oplus$};
\node[val] (bias) at (4.6, -1.0) {$b$};
\draw[conn, shorten >=3pt, shorten <=4pt] (sum) -- (add);
\draw[conn, shorten >=3pt, shorten <=2pt] (bias) -- (add);

% --- Activation function ---
\node[actnode] (act) at (6.4, 0) {$\sigma$};
\draw[conn, shorten >=4pt, shorten <=3pt] (add) -- (act) node[mathlbl, midway, above] {$z$};

% --- Output value ---
\node[val] (out) at (8.8, 0) {$a$};
\draw[conn, shorten >=6pt, shorten <=4pt] (act) -- (out);

% --- Section labels above ---
\node[seclbl] at (-0.6, 2.8) {inputs};
\node[seclbl] at (2.8, 2.8) {weighted sum};
\node[seclbl] at (6.4, 2.8) {activation};
\node[seclbl] at (8.8, 2.8) {output};

\end{tikzpicture}
\caption{A single neuron. Each arrow carries its input multiplied
  by the weight shown ($w$); the products are summed, a bias ($b$)
  is added, and the activation function produces the output.}
\label{fig:neuron}
\end{figure}
