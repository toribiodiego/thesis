\section{Part 2: Rainbow Capability Comparison (Thesis 4.2)}

\noindent These figures answer: \textit{Does SPR need a capable
base agent to produce useful representations?}
Reproduction: Appendix~\ref{app:perf-figs}.

\subsection*{B1. Learning Curves -- Rainbow Comparison}

\noindent\textit{Purpose.} Does SPR need a stronger base agent
to be effective? Compares learning trajectories for DQN+SPR,
Rainbow, and Rainbow+SPR on the same six games -- if SPR only
helps Rainbow, the base agent's learning infrastructure matters
more than the representation objective itself.

\begin{itemize}[leftmargin=2em, itemsep=0.1em, topsep=0.3em]
  \item Three conditions: DQN+SPR, Rainbow, Rainbow+SPR
  \item 6 subplots (one per game), same format as A1
  \item Shaded bands: $\pm 1$ std or cross-seed confidence interval
  \item Status: Crazy Climber shown (seed 42); full 6-game
    grid after batch runs
\end{itemize}

\begin{figure}[H]
\centering

\caption{Crazy Climber (seed 42). Rainbow dominates DQN+SPR by
$\sim$5$\times$. Rainbow+SPR is comparable to Rainbow alone on
this game.}
\end{figure}

\subsection*{B2. SPR Lift Comparison}

\noindent\textit{Purpose.} How much does SPR improve DQN compared

    (Rainbow+SPR $-$ Rainbow)

\section{Part 3: Interpretability Analysis (Thesis 4.3)}
% ================================================================

\noindent These figures answer: \textit{What do the learned
representations actually encode, and does that change between
DQN and Rainbow?}

\noindent All figures in this section require the feature
extraction and probing pipeline. Checkpoints with
\texttt{spr\_state} will be available after batch runs.
Reproduction: Appendix~\ref{app:interp-figs}.

\subsection*{C1. Probe Accuracy}

\noindent\textit{Purpose.} Do SPR-trained encoders learn more
about the game state than vanilla encoders? Linear probes on
frozen encoder outputs test whether features like player position
and score are decodable -- higher accuracy means the encoder
captures more game-relevant structure.

\begin{itemize}[leftmargin=2em, itemsep=0.1em, topsep=0.3em]
  \item Grouped bars: F1 per game, per condition, per probe target
  \item Probe targets: player position, score, lives (from ALE RAM)
  \item All 6 conditions compared on same axes
  \item Status: pending -- requires feature extraction and
    probing pipeline
\end{itemize}

\vspace{0.5em}
\subsection*{C2. Probe Accuracy Over Training}

\noindent\textit{Purpose.} When during training do the
representations become useful? Probing at multiple checkpoints
shows whether feature quality improves steadily, plateaus early,
or only emerges late -- and whether DQN and Rainbow develop
useful representations on different timelines.

\begin{itemize}[leftmargin=2em, itemsep=0.1em, topsep=0.3em]
  \item x = checkpoint step, y = probe F1, lines per condition
  \item One plot per game, all 6 conditions overlaid
  \item 10 checkpoints (every 40K frames) for new runs,
    4 checkpoints for existing DQN/+Aug runs
  \item Status: pending -- requires feature extraction and
    probing at each checkpoint
\end{itemize}

\vspace{0.5em}
\subsection*{C3. Latent Space Visualization}

\noindent\textit{Purpose.} What does the representation space
look like? PCA and t-SNE projections reveal whether SPR produces
clusters that correspond to game-relevant features (e.g., player
position) or whether the encoder learns unstructured
representations regardless of the training objective.

\begin{itemize}[leftmargin=2em, itemsep=0.1em, topsep=0.3em]
  \item 2D scatter plots colored by probe target
  \item Grid layout: rows = conditions, columns = games
  \item PCA for reproducibility (primary), t-SNE for local
    structure (supplementary)
  \item Status: pending -- requires feature extraction
\end{itemize}

\subsection*{C5. Metric Correlation}

\noindent\textit{Purpose.} Which Rainbow training metrics track
representation quality? Correlating metrics like distributional
loss, priority entropy, and importance sampling weights with
probe accuracy identifies which of Rainbow's components are
most connected to what the encoder learns.

\begin{itemize}[leftmargin=2em, itemsep=0.1em, topsep=0.3em]
  \item Heatmap: rows = Rainbow metrics, columns = games
  \item Cells = Pearson/Spearman correlation with probe F1
  \item Strongest correlations point to candidates for targeted
    ablation
  \item Status: pending -- requires probe accuracy timeseries
    and Rainbow training logs
\end{itemize}

\vspace{0.5em}

\subsection*{C6. Representation Similarity}

\noindent\textit{Purpose.} Does SPR fundamentally change what the
encoder learns, or does it refine similar features? Comparing
representations across conditions (e.g., DQN vs DQN+SPR vs
Rainbow vs Rainbow+SPR) using CKA or similar metrics reveals
whether the interventions produce structurally different
representations or just sharper versions of the same ones.

\begin{itemize}[leftmargin=2em, itemsep=0.1em, topsep=0.3em]
  \item Similarity matrix: conditions $\times$ conditions,
    cells = CKA score
  \item One matrix per game, or averaged across games
  \item High similarity = same features learned, low = different
    representations
  \item Status: pending -- requires feature extraction, computed
    from extracted encoder outputs
\end{itemize}

\vspace{0.5em}

\begin{itemize}[leftmargin=2em, itemsep=0.1em, topsep=0.3em]

    condition
  \item Grouped bars: one group per target, bars per condition
  \item Status: pending -- produced alongside C1 from the same
    probe results
\end{itemize}

\section{Regeneration}

\subsection{Downloading run data}
\label{app:download}

\begin{verbatim}
bash scripts/pull-run.sh --all              # all runs
bash scripts/pull-run.sh --group base       # vanilla DQN
bash scripts/pull-run.sh --group spr        # DQN+SPR
bash scripts/pull-run.sh --group rainbow    # Rainbow
bash scripts/pull-run.sh --all --no-checkpoints  # skip .pt files
bash scripts/pull-run.sh --list             # list available
\end{verbatim}

Run directories are documented in
\texttt{experiments/dqn\_atari/runs/INDEX.md}.

\subsection{Performance figures (A, B, D1--D7)}
\label{app:perf-figs}

\begin{verbatim}
python scripts/plot_learning_curves.py  # learning curves
python scripts/plot_results.py          # score tables, bar charts
\end{verbatim}

\noindent Both scripts read \texttt{evaluations.csv} and
\texttt{training\_steps.csv} from local run directories.

\begin{verbatim}
python scripts/reeval_checkpoints.py --device cuda  # fast
python scripts/reeval_checkpoints.py --record-video # with video
python scripts/reeval_checkpoints.py --force        # re-run all
\end{verbatim}

\noindent Run on Colab T4, not locally. See
\texttt{reports/eval-pipeline.tex} for details.
