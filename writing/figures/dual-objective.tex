% Cuboid macro (from nature-cnn.tex)
\newcommand{\cuboid}[7]{%
    \fill[#5, draw=black!35, line join=round]
        (#1, {-#3/2}, 0) -- ({#1+#2}, {-#3/2}, 0) --
        ({#1+#2}, {#3/2}, 0) -- (#1, {#3/2}, 0) -- cycle;
    \fill[#6, draw=black!35, line join=round]
        (#1, {#3/2}, 0) -- ({#1+#2}, {#3/2}, 0) --
        ({#1+#2}, {#3/2}, #4) -- (#1, {#3/2}, #4) -- cycle;
    \fill[#7, draw=black!35, line join=round]
        ({#1+#2}, {-#3/2}, 0) -- ({#1+#2}, {#3/2}, 0) --
        ({#1+#2}, {#3/2}, #4) -- ({#1+#2}, {-#3/2}, #4) -- cycle;
}

\caption{Adding self-supervised learning to DQN.
  (a)~In standard DQN, the Q-learning loss is the sole signal
  shaping the encoder.
  (b)~Self-supervised learning adds a second objective: the encoder
  receives gradients from both $\mathcal{L}_{\text{Q}}$ (action
  values) and $\lambda\mathcal{L}_{\text{SSL}}$ (dynamics),
  producing richer representations.}
\label{fig:dual-objective}
